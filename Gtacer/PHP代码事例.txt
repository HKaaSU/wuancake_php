使用global访问全局变量

<?php
$x=5;
$y=10;

function myTest() {
  global $x,$y;
  $y=$x+$y;
}

myTest();
echo $y; // 输出 15
?>

for循环
<?php
$a=1;
for($a;$a<=10;$a++){
    echo $a;
}

?>


while循环
<?php
    $i=1;
    while($i<10){
        echo $i;
        $i++;
    }
?>



do while 循环
<?php
	$i=0;
	do{
		echo $i;
		$i++;
}while($i<10);
